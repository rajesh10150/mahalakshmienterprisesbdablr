// GET — fetch all plots from Sheet
async function sheetsGet() {
  const r = await fetch(`${SHEET_API}?action=getPlots`, { method: 'GET' });
  if (!r.ok) throw new Error('HTTP ' + r.status);
  const data = await r.json();
  if (data.status !== 'ok') throw new Error(data.message || 'Sheet error');
  return data.plots;
}

// POST — write plots back to Sheet
async function sheetsPost(action, payload) {
  const r = await fetch(SHEET_API, {
    method: 'POST',
    headers: { 'Content-Type': 'text/plain' }, // Apps Script requires text/plain for CORS
    body: JSON.stringify({ action, ...payload })
  });
  if (!r.ok) throw new Error('HTTP ' + r.status);
  const data = await r.json();
  if (data.status !== 'ok') throw new Error(data.message || 'Sheet error');
  return data;
}

// ── CACHE HELPERS ──
function cacheLoad() {
  try {
    const c = localStorage.getItem(CACHE_KEY);
    return c ? JSON.parse(c) : null;
  } catch(e) { return null; }
}

function cacheSave(plots) {
  try {
    localStorage.setItem(CACHE_KEY, JSON.stringify(plots));
    localStorage.setItem(SYNC_KEY, Date.now().toString());
  } catch(e) {}
}

function cacheStale() {
  const t = parseInt(localStorage.getItem(SYNC_KEY) || '0');
  return Date.now() - t > SYNC_INTERVAL;
}

// ── SET SYNC STATUS BADGE ──
function setSyncStatus(s, msg) {
  syncStatus = s;
  const badge = document.getElementById('syncBadge');
  if (!badge) return;
  const states = {
    syncing: { icon:'⟳', text:'Syncing…',    cls:'sync-syncing' },
    ok:      { icon:'✅', text:'Sheets Live', cls:'sync-ok'      },
    error:   { icon:'⚠️', text:'Sheet Error', cls:'sync-error'   },
    offline: { icon:'💾', text:'Offline Mode', cls:'sync-offline' },
    idle:    { icon:'',   text:'',            cls:''             }
  };
  const st = states[s] || states.idle;
  badge.textContent  = st.icon + ' ' + (msg || st.text);
  badge.className    = 'sync-badge ' + st.cls;
  badge.style.display = s === 'idle' ? 'none' : 'inline-flex';
}

// ── LOAD PLOTS (smart: cache first, then Sheet) ──
async function loadPlots(force = false) {
  setSyncStatus('syncing');

  // 1. Immediately show cache or defaults so UI isn't blank
  const cached = cacheLoad();
  if (cached && cached.length) {
    livePlots = cached;
    refreshAllViews();
  } else {
    livePlots = DEFAULT_PLOTS;
    refreshAllViews();
  }

  // 2. If Sheet not configured → use localStorage admin panel
  if (!sheetReady()) {
    const local = localStorage.getItem('mahalakshmi_plots');
    if (local) livePlots = JSON.parse(local);
    else livePlots = DEFAULT_PLOTS;
    refreshAllViews();
    setSyncStatus('offline', '💾 Local Mode');
    return;
  }

  // 3. Skip if cache is fresh and not forced
  if (!force && !cacheStale() && cached && cached.length) {
    setSyncStatus('ok', '✅ Sheets Live');
    return;
  }

  // 4. Fetch from Google Sheets
  try {
    const plots = await sheetsGet();
    livePlots = plots;
    cacheSave(plots);
    refreshAllViews();
    setSyncStatus('ok');
  } catch(e) {
    console.warn('Sheet fetch failed:', e.message);
    setSyncStatus('error', '⚠️ Using Cache');
  }
}

  if (!sheetReady()) {
    // local only
    const plots = cacheLoad() || DEFAULT_PLOTS;
    plots.push(plot);
    cacheSave(plots);

    localStorage.setItem('mahalakshmi_plots', JSON.stringify(plots));
    return;
  }
  setSyncStatus('syncing', '⟳ Saving…');
  await sheetsPost('addPlot', { plot });
  await loadPlots(true);
}

async function sheetsUpdatePlot(plot) {
  if (!sheetReady()) {
    const plots = cacheLoad() || DEFAULT_PLOTS;
    const idx = plots.findIndex(p => p.id === plot.id);
    if (idx !== -1) plots[idx] = plot;
    cacheSave(plots);
    localStorage.setItem('mahalakshmi_plots', JSON.stringify(plots));
    return;
  }
  setSyncStatus('syncing', '⟳ Updating…');
  await sheetsPost('updatePlot', { plot });
  await loadPlots(true);
}

async function sheetsDeletePlot(id) {
  if (!sheetReady()) {
    const plots = (cacheLoad() || DEFAULT_PLOTS).filter(p => p.id !== id);
    cacheSave(plots);
    localStorage.setItem('mahalakshmi_plots', JSON.stringify(plots));
    return;
  }
  setSyncStatus('syncing', '⟳ Deleting…');
  await sheetsPost('deletePlot', { id });
  await loadPlots(true);
}

  await sheetsPost('toggleStatus', { id });

}

const DEFAULT_PLOTS = JSON.parse(JSON.stringify(properties)); // snapshot of hardcoded defaults

function getPlots() {
  try {
    const saved = localStorage.getItem('mahalakshmi_plots');
    if (saved) return JSON.parse(saved);
  } catch(e) {}
  // First run — seed from hardcoded list
  savePlots(DEFAULT_PLOTS);
  return DEFAULT_PLOTS;
}

function savePlots(list) {
  try {
    localStorage.setItem('mahalakshmi_plots', JSON.stringify(list));
  } catch(e) { console.error('Save failed', e); }
}

function getNextId() {
  const plots = getPlots();
  return plots.length ? Math.max(...plots.map(p => p.id)) + 1 : 1;
}

// Submits to both FormSubmit.co (email) and Google Sheets
async function submitToAll(data) {
  const results = await Promise.allSettled([
    // ① Email via FormSubmit.co
    fetch(`https://formsubmit.co/ajax/${YOUR_EMAIL}`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json', 'Accept': 'application/json' },
      body: JSON.stringify(data)
    }),
    // ② Save lead to Google Sheets Leads tab
    sheetsAddLead(data)
  ]);
  const emailResult = results[0];
  if (emailResult.status === 'fulfilled') {
    const res = emailResult.value;
    if (res.ok) return true;
  }
  return false;
}

function filterProperties() {
  const s = document.getElementById('filterSize').value;
  const b = document.getElementById('filterBlock').value;
  const f = document.getElementById('filterFacing').value;
  const st = document.getElementById('filterStatus').value;
  const ty = document.getElementById('filterType').value;
  const filtered = livePlots.filter(p =>
    (!s || p.size === s) &&
    (!b || p.block.includes(b)) &&
    (!f || p.facing === f) &&
    (!st || p.status === st) &&
    (!ty || p.type === ty)
  );
  renderProperties(filtered);
  renderInlineTable(filtered);
  updateLiveCount(filtered);
}

let roiYears = 5;

function fmtINR(n) {
  if (n >= 10000000) return '₹' + (n / 10000000).toFixed(2) + ' Cr';
  if (n >= 100000)   return '₹' + (n / 100000).toFixed(1) + ' L';
  return '₹' + Math.round(n).toLocaleString('en-IN');
}

function setEMIPrice(val) {
  document.getElementById('emiPrice').value = val;
  calcEMI();
}

function calcEMI() {
  const price   = parseFloat(document.getElementById('emiPrice').value) || 0;
  const downPct = parseFloat(document.getElementById('emiDown').value) / 100;
  const tenure  = parseInt(document.getElementById('emiTenure').value);
  const rate    = parseFloat(document.getElementById('emiRate').value);

  const downAmt = price * downPct;
  const loan    = price - downAmt;
  const r       = rate / 100 / 12;
  const n       = tenure * 12;
  const emi     = r === 0 ? loan / n : (loan * r * Math.pow(1 + r, n)) / (Math.pow(1 + r, n) - 1);
  const total   = emi * n;
  const intAmt  = total - loan;

  document.getElementById('downPctLabel').textContent = Math.round(downPct * 100) + '%';
  document.getElementById('downAmtLabel').textContent = fmtINR(downAmt);
  document.getElementById('tenureLabel').textContent  = tenure + ' year' + (tenure > 1 ? 's' : '');
  document.getElementById('rateLabel').textContent    = rate.toFixed(1) + '%';
  document.getElementById('emiMonthly').textContent   = fmtINR(emi) + '/mo';
  document.getElementById('emiLoan').textContent      = fmtINR(loan);
  document.getElementById('emiDownAmt').textContent   = fmtINR(downAmt);
  document.getElementById('emiInterest').textContent  = fmtINR(intAmt);
  document.getElementById('emiTotal').textContent     = fmtINR(total);

  // Donut chart
  const totalCirc = 314;
  const principalRatio = loan / (loan + intAmt);
  const pDash = totalCirc * principalRatio;
  const iDash = totalCirc * (1 - principalRatio);
  const iOffset = totalCirc - pDash;

  const pArc = document.getElementById('emiPrincipalArc');
  const iArc = document.getElementById('emiInterestArc');
  pArc.setAttribute('stroke-dasharray', pDash + ' ' + (totalCirc - pDash));
  pArc.setAttribute('stroke-dashoffset', 0);
  iArc.setAttribute('stroke-dasharray', iDash + ' ' + (totalCirc - iDash));
  iArc.setAttribute('stroke-dashoffset', -pDash);
}

function setROIPrice(val) {

function setROIYear(y, btn) {
  roiYears = y;
  document.querySelectorAll('.roi-yr-btn').forEach(b => b.classList.remove('active'));
  btn.classList.add('active');
  calcROI();
}

function calcROI() {
  const price = parseFloat(document.getElementById('roiPrice').value) || 0;
  const rate  = parseFloat(document.getElementById('roiRate').value) / 100;

  document.getElementById('roiRateLabel').textContent = Math.round(rate * 100) + '%';
  document.getElementById('roiToday').textContent     = fmtINR(price);
  document.getElementById('roiYearLabel').textContent = 'In ' + roiYears + ' Year' + (roiYears > 1 ? 's' : '');

  const future = price * Math.pow(1 + rate, roiYears);
  const profit = future - price;
  const pct    = ((future - price) / price * 100).toFixed(0);

  document.getElementById('roiFuture').textContent  = fmtINR(Math.round(future));
  document.getElementById('roiProfit').textContent  = fmtINR(Math.round(profit));
  document.getElementById('roiReturn').textContent  = '+' + pct + '%';
  document.getElementById('roiCAGR').textContent    = Math.round(rate * 100) + '%';

  // Build year bars
  const maxVal = future;
  let bars = '';
  const yrs = roiYears <= 3 ? roiYears : (roiYears === 5 ? 5 : 10);
  for (let i = 1; i <= yrs; i++) {
    const v = price * Math.pow(1 + rate, i);
    const pct = Math.round((v / maxVal) * 100);
    bars += `<div class="roi-bar-row">
      <div class="roi-bar-label">Yr ${i}</div>
      <div class="roi-bar-bg">
        <div class="roi-bar-fill" style="width:${pct}%">
          <span>${fmtINR(Math.round(v))}</span>
        </div>
      </div>
    </div>`;
  }
  document.getElementById('roiBarWrap').innerHTML = bars;
}

      btn.style.background = '';
      btn.style.borderColor = '';
      btn.style.color = '';
    }, 2000);
  });
}

async function saveAdminForm() {
  const siteNo  = document.getElementById('fSiteNo').value.trim();
  const block   = document.getElementById('fBlock').value.trim();
  const sector  = document.getElementById('fSector').value.trim();
  const facing  = document.getElementById('fFacing').value;
  const size    = document.getElementById('fSize').value;
  const sqft    = parseInt(document.getElementById('fSqft').value) || 0;
  const type    = document.getElementById('fType').value;
  const price   = parseFloat(document.getElementById('fPrice').value) || 0;
  const status  = document.getElementById('fStatus').value;
  const remarks = document.getElementById('fRemarks').value.trim();

  if (!siteNo || !block || !sector || !price) {
    alert('Please fill in: Site No, Block, Sector, and Price');
    return;
  }

  const saveBtn = document.querySelector('.adm-save');
  saveBtn.textContent = '⟳ Saving…';
  saveBtn.disabled = true;

  try {
    if (editingId !== null) {
      const plots = getPlots();
      const existing = plots.find(p => p.id === editingId);
      const updated = { ...existing, siteNo, block, sector, facing, size, sqft, type, price, status, remarks };
      await sheetsUpdatePlot(updated);
      showAdminToast(`Site ${siteNo} updated ✅`);
    } else {
      const newPlot = { id: getNextId(), siteNo, block, sector, facing, size, sqft, type, price, status, remarks };
      await sheetsAddPlot(newPlot);
      showAdminToast(`Site ${siteNo} added ✅`);
    }
    livePlots = getPlots();
    closeAdminForm();
    renderAdminTable();
    refreshAllViews();
  } catch(e) {
    showAdminToast('❌ Error: ' + e.message);
    setSyncStatus('error');
  } finally {
    saveBtn.textContent = '💾 Save Plot';
    saveBtn.disabled = false;
  }
}

  const text = await file.text();
  const rows = text.trim().split('\n');
  const hdrs = rows[0].split(',').map(h => h.trim().replace(/^"|"$/g,'').toLowerCase());
  const get = (vals, keys) => { for(const k of keys){const i=hdrs.indexOf(k);if(i!==-1&&vals[i])return vals[i];} return ''; };

  const plots = []; let nid = getNextId();
  for (let i = 1; i < rows.length; i++) {
    const v = rows[i].split(',').map(x => x.trim().replace(/^"|"$/g,''));
    if (v.length < 3) continue;
    plots.push({
      id: nid++,
      siteNo:  get(v,['siteno','site no','site_no']) || `IMP-${i}`,
      block:   get(v,['block']) || 'Unknown',
      sector:  get(v,['sector']) || '—',
      facing:  get(v,['facing']) || 'North',
      size:    get(v,['size']) || '30×40',
      sqft:    parseInt(get(v,['sqft','area'])) || 1200,
      type:    get(v,['type']) || 'Residential',
      price:   parseFloat(get(v,['price'])) || 0,
      status:  get(v,['status']) || 'Available',
      remarks: get(v,['remarks','notes'])
    });
  }
  if (!plots.length) { showAdminToast('❌ No valid plots in CSV'); return; }
  if (!confirm(`Import ${plots.length} plots? They will be ADDED to your existing list.`)) return;
  showAdminToast(`⟳ Importing ${plots.length} plots…`);
  for (const p of plots) await sheetsAddPlot(p);
  livePlots = getPlots(); renderAdminTable(); refreshAllViews();
  showAdminToast(`✅ Imported ${plots.length} plots!`);
  event.target.value = '';
}

// ── SETUP STATUS CHECK ──
async function checkSetupStatus() {
  const icon = document.getElementById('setupStatusIcon');
  const text = document.getElementById('setupStatusText');
  if (!icon) return;
  if (!sheetReady()) {
    icon.textContent = '⚠️';
    text.innerHTML = '<strong style="color:#f87171;">Not connected.</strong> Paste your Apps Script URL in the HTML file to enable sync.';
    return;
  }
  icon.textContent = '⟳'; text.textContent = 'Testing connection…';
  try {
    const plots = await sheetsGet();
    icon.textContent = '✅';
    text.innerHTML = `<strong style="color:#4ade80;">Connected!</strong> ${plots.length} plots live in Google Sheets.`;
  } catch(e) {
    icon.textContent = '❌';
    text.innerHTML = `<strong style="color:#f87171;">Failed:</strong> ${e.message}`;
  }
}
